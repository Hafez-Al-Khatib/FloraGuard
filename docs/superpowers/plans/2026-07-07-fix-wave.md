`auth.py` — full new content:

```python
"""Bearer-token auth returning a scoped Principal.

* ``require_auth``  — admin token OR a per-device token. Returns a
  ``Principal`` so routes can check *which* device is calling.
* ``require_admin`` — admin token only (ops/control routes).

A device principal may only act on its own node_id — enforced by the routes
via ``principal.assert_node(node_id)``.
"""
import secrets
from dataclasses import dataclass
from typing import Literal

from fastapi import Depends, HTTPException, Request, status
from fastapi.security import HTTPBearer, HTTPAuthorizationCredentials

from config import get_settings

security = HTTPBearer(auto_error=False)


@dataclass(frozen=True)
class Principal:
    kind: Literal["admin", "device"]
    node_id: str | None = None

    @property
    def is_admin(self) -> bool:
        return self.kind == "admin"

    def assert_node(self, node_id: str) -> None:
        """403 unless admin or the device that owns ``node_id``."""
        if self.is_admin or self.node_id == node_id:
            return
        raise HTTPException(
            status_code=status.HTTP_403_FORBIDDEN,
            detail="Token is not authorized for this node.",
        )


def _bearer(credentials: HTTPAuthorizationCredentials | None) -> str:
    if credentials is None or credentials.scheme.lower() != "bearer":
        raise HTTPException(
            status_code=status.HTTP_401_UNAUTHORIZED,
            detail="Missing or invalid Authorization header. Use 'Bearer <token>'.",
        )
    token = credentials.credentials.strip()
    if not token:
        raise HTTPException(
            status_code=status.HTTP_401_UNAUTHORIZED,
            detail="Empty bearer token.",
        )
    return token


async def require_auth(
    request: Request,
    credentials: HTTPAuthorizationCredentials = Depends(security),
) -> Principal:
    settings = get_settings()
    token = _bearer(credentials)

    if secrets.compare_digest(token, settings.api_auth_token):
        return Principal(kind="admin")

    cache = getattr(request.app.state, "cache", None)
    if cache is not None:
        try:
            node_id = await cache.verify_device_token(token)
            if node_id:
                return Principal(kind="device", node_id=node_id)
        except Exception:  # pragma: no cover - never 500 on auth
            pass

    raise HTTPException(
        status_code=status.HTTP_403_FORBIDDEN,
        detail="Invalid authentication token.",
    )


async def require_admin(
    credentials: HTTPAuthorizationCredentials = Depends(security),
) -> Principal:
    settings = get_settings()
    token = _bearer(credentials)
    if not secrets.compare_digest(token, settings.api_auth_token):
        raise HTTPException(
            status_code=status.HTTP_403_FORBIDDEN,
            detail="Admin token required for this operation.",
        )
    return Principal(kind="admin")
```

    async def note_node_seen(
        self, node_id: str, default_profile: dict | None = None
    ) -> None:
        """Idempotent 'this node exists' marker for ingest paths.

        Unlike register_node this never overwrites an existing profile and

                    await r.hsetnx(f"nodes:profile:{node_id}", k, str(v))

    EVENT_TYPES = ("telemetry", "alert", "detection", "actuator", "online")

    async def emit_event(self, event_type: str, node_id: str, payload: dict) -> None:

  double? moisture, double? temperature, double? ec, double? batteryPct,
  String? timestamp, String? resetReason, int? freeHeap, String? lastSeen,

/// Applies one typed SSE event. Only `telemetry` events refresh liveness —
/// a server-generated actuator/detection event must not make a dead node
/// look alive.
TelemetrySnapshot applyEvent(String type, Map<String, dynamic> payload) {
  final now = DateTime.now().toUtc().toIso8601String();
  switch (type) {
    case 'telemetry':
      return copyWith(
        moisture: (payload['moisture'] as num?)?.toDouble(),
        temperature: (payload['temperature'] as num?)?.toDouble(),
        ec: (payload['ec'] as num?)?.toDouble(),
        batteryPct: (payload['battery_pct'] as num?)?.toDouble(),
        lastSeen: now,
        timestamp: now,
        updateTick: updateTick + 1,
      );
    case 'detection':
      return copyWith(
        detectionIssue: payload['issue'] as String?,
        detectionConfidence: (payload['confidence'] as num?)?.toDouble(),
        detectionAt: payload['at'] as String?,
        updateTick: updateTick + 1,
      );
    case 'actuator':
      return copyWith(
        actuatorOn: payload['on'] as bool?,
        actuatorReason: payload['reason'] as String?,
        actuatorBound: payload['bound'] as String?,
        actuatorMode: payload['mode'] as String?,
        updateTick: updateTick + 1,
      );
    case 'online':
      return copyWith(
        profile: payload['profile'] as Map<String, dynamic>?,
      );
    default:
      return this; // unknown/alert types don't mutate the snapshot
  }
}
